Vim/app.go
package main

import (
    "net/http"
    "github.com/goanywhere/rex"
)

func main() {
    app := rex.New()
    app.Get("/", func(w http.ResponseWriter, r *http.Request) {
        ctx := rex.NewContext(w, r)
        ctx.String("Hello ♔ Rex")
    })
    app.Serve()
}
